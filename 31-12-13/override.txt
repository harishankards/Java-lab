class fig
{	
	double dim1`,dim2;
	fig(double a,double b)
	{
		dim1=a;
		dim2=b;
	}
	double area()
	{	
		System.out.println("Inside area for fig:");
		return (dim1*dim2);
	}
}
class rectangle extends fig
{
	rectangle(double a,double b)
	{
		super(a,b);
	}
	double area()
	{
		System.out.println("Inside area or rectangle");
		return (dim1*dim2);
	}
}
